001package com.monochromeroad.grails.plugins.xwiki;
002
003import groovy.lang.Closure;
004import org.xwiki.rendering.block.XDOM;
005import org.xwiki.rendering.parser.ParseException;
006import org.xwiki.rendering.parser.StreamParser;
007import org.xwiki.rendering.renderer.Renderer;
008import org.xwiki.rendering.syntax.Syntax;
009import org.xwiki.rendering.transformation.Transformation;
010
011import java.io.Reader;
012import java.util.*;
013
014/**
015 * XWiki Rendering System -- streaming based
016 *
017 * <p>
018 * If you use some macros or transformations, it needs to create a XDOM that represents the whole document structure in a memory.
019 * </p>
020 *
021 * <dl>
022 *     <dt>Default Syntax</dt>
023 *     <dd>xwiki/2.1</dd>
024 *     <dt>Output</dt>
025 *     <dd>xhtml/1.0</dd>
026 * </dl>
027 *
028 * @author Masatoshi Hayashi
029 */
030public class XWikiStreamRenderer extends XWikiRenderingSystem {
031
032    public XWikiStreamRenderer(XWikiComponentManager componentManager, XWikiConfigurationProvider configuration) {
033        super(componentManager, configuration);
034    }
035
036    /**
037     * For the Grails Default XWiki Rendering System.
038     * It need to be initialized after construction.
039     */
040    XWikiStreamRenderer() {}
041
042    /**
043     * XWiki rendering
044     *
045     * @param source source text reader
046     * @param inputSyntax inputSyntax
047     * @param outputSyntax outputSyntax
048     * @param callback a callback function processed on parsing a text
049     */
050    public void render(Reader source, Syntax inputSyntax, Syntax outputSyntax, Closure callback) {
051        if (configurationProvider.isMacrosEnabled()) {
052            renderOnXDOM(source, inputSyntax, outputSyntax, Collections.<Transformation>emptyList(), callback);
053        } else {
054            renderOnStream(source, inputSyntax, outputSyntax, callback);
055        }
056    }
057
058    /**
059     * XWiki XHTML rendering
060     *
061     * @param source source text reader
062     * @param inputSyntax inputSyntax
063     * @param callback a callback function processed on parsing a text
064     */
065    public void render(Reader source, Syntax inputSyntax, Closure callback) {
066        Syntax outputSyntax = configurationProvider.getDefaultOutputSyntax();
067        render(source, inputSyntax, outputSyntax, callback);
068    }
069
070    /**
071     * XWiki XHTML rendering using the default syntax
072     *
073     * @param source source text reader
074     * @param callback a callback function processed on parsing a text
075     */
076    public void render(Reader source, Closure callback) {
077        Syntax inputSyntax = configurationProvider.getDefaultInputSyntax();
078        Syntax outputSyntax = configurationProvider.getDefaultOutputSyntax();
079        render(source, inputSyntax, outputSyntax, callback);
080    }
081
082    /**
083     * XWiki rendering
084     *
085     * @param source source text reader
086     * @param inputSyntax inputSyntax
087     * @param outputSyntax outputSyntax
088     * @param transformations transform parameters
089     * @param callback a callback function processed on parsing a text
090     */
091    public void render(Reader source, Syntax inputSyntax, Syntax outputSyntax, List<Transformation> transformations, Closure callback) {
092        if (configurationProvider.isMacrosEnabled() || !transformations.isEmpty()) {
093            renderOnXDOM(source, inputSyntax, outputSyntax, transformations, callback);
094        } else {
095            renderOnStream(source, inputSyntax, outputSyntax, callback);
096        }
097    }
098
099    /**
100     * XWiki XHTML rendering
101     *
102     * @param source source text reader
103     * @param inputSyntax inputSyntax
104     * @param transformations transform parameters
105     * @param callback a callback function processed on parsing a text
106     */
107    public void render(Reader source, Syntax inputSyntax, List<Transformation> transformations, Closure callback) {
108        Syntax outputSyntax = configurationProvider.getDefaultOutputSyntax();
109        render(source, inputSyntax, outputSyntax, transformations, callback);
110    }
111
112    /**
113     * XWiki XHTML rendering using the default syntax
114     *
115     * @param source source text reader
116     * @param transformations transform parameters
117     * @param callback a callback function processed on parsing a text
118     */
119    public void render(Reader source, List<Transformation> transformations, Closure callback) {
120        Syntax inputSyntax = configurationProvider.getDefaultInputSyntax();
121        Syntax outputSyntax = configurationProvider.getDefaultOutputSyntax();
122        render(source, inputSyntax, outputSyntax, transformations, callback);
123    }
124
125    private void renderOnXDOM(Reader source, Syntax inputSyntax, Syntax outputSyntax, List<Transformation> transformations, Closure callback) {
126        XDOM xdom = buildXDOM(source, inputSyntax);
127        transform(xdom, inputSyntax, transformations);
128        applyRenderer(xdom, outputSyntax, new XWikiCallbackPrinter(callback));
129    }
130
131    private void renderOnStream(Reader source, Syntax inputSyntax, Syntax outputSyntax, Closure callback) {
132        StreamParser streamParser = componentManager.getInstance(StreamParser.class, inputSyntax.toIdString());
133        XWikiCallbackPrinter printer = new XWikiCallbackPrinter(callback);
134        Renderer renderer = createRenderer(outputSyntax, printer);
135        try {
136            streamParser.parse(source, renderer);
137        } catch (ParseException e) {
138            throw new IllegalStateException(e);
139        }
140    }
141}
